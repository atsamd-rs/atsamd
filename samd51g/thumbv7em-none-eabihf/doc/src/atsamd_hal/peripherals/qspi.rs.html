32    /// Enable the clocks for the qspi peripheral in single data rate mode
33    /// assuming 120mhz system clock, for 4mhz spi mode 0 operation.
34    #[allow(clippy::too_many_arguments)]
35    pub fn new(
36        mclk: &mut Mclk,
37        qspi: pac::Qspi,
38        _sck: impl AnyPin<Id = PB10>,
39        _cs: impl AnyPin<Id = PB11>,
40        _io0: impl AnyPin<Id = PA08>,
41        _io1: impl AnyPin<Id = PA09>,
42        _io2: impl AnyPin<Id = PA10>,
43        _io3: impl AnyPin<Id = PA11>,
44    ) -> Qspi<OneShot> {
45        mclk.apbcmask().modify(|_, w| w.qspi_().set_bit());
46        // Enable the clocks for the qspi peripheral in single data rate mode.
47        mclk.ahbmask().modify(|_, w| {
48            w.qspi_().set_bit();
49            w.qspi_2x_().clear_bit()
50        });
51
52        let _sck = _sck.into().into_alternate();
53        let _cs = _cs.into().into_alternate();
54        let _io0 = _io0.into().into_alternate();
55        let _io1 = _io1.into().into_alternate();
56        let _io2 = _io2.into().into_alternate();
57        let _io3 = _io3.into().into_alternate();
58
59        qspi.ctrla().write(|w| w.swrst().set_bit());
60        qspi.baud().write(|w| unsafe {
61            // TODO get system clock value instead of hardcoding
62            //(120_000_000u32 / 4_000_000u32) = 30 = BAUD + 1
63            // BAUD = 29
64            w.baud().bits(29); // 4Mhz
65            // SPI MODE 0
66            w.cpol().clear_bit();
67            w.cpha().clear_bit()
68        });
69
70        qspi.ctrlb().write(|w| {
71            w.mode().memory();
72            w.csmode().noreload();
73            w.csmode().lastxfer();
74            w.datalen()._8bits()
75        });
76
77        qspi.ctrla().modify(|_, w| w.enable().set_bit());
78
79        Self {
80            qspi,
81            _sck,
82            _cs,
83            _io0,
84            _io1,
85            _io2,
86            _io3,
87            _mode: PhantomData,
88        }
89    }

92    pub fn run_command(&self, command: Command) -> Result<(), Error> {
93        match command {
94            //TODO verify this list of commands
95            Command::WriteEnable
96            | Command::WriteDisable
97            | Command::Reset
98            | Command::EnableReset => (),
99            _ => return Err(Error::CommandFunctionMismatch),
100        }
101
102        let tfm = TransferMode {
103            instruction_enable: true,
104            ..TransferMode::default()
105        };
106        unsafe {
107            self.run_read_instruction(command, tfm, 0, &mut [], true);
108        }
109        Ok(())
110    }

285// (Mostly internal) methods available in any mode.
286impl<MODE> Qspi<MODE> {
287    unsafe fn finalize(&self) {
288        self.qspi.ctrla().write(|w| {
289            w.enable().set_bit();
290            w.lastxfer().set_bit()
291        });
292
293        while self.qspi.intflag().read().instrend().bit_is_clear() {}
294        self.qspi.intflag().write(|w| w.instrend().set_bit());
295        while self.qspi.intflag().read().csrise().bit_is_clear() {}
296        self.qspi.intflag().write(|w| w.csrise().set_bit());
297    }
298
299    unsafe fn run_write_instruction(
300        &self,
301        command: Command,
302        tfm: TransferMode,
303        addr: u32,
304        buf: &[u8],
305    ) {
306        unsafe {
307            if command == Command::EraseSector || command == Command::EraseBlock {
308                self.qspi.instraddr().write(|w| w.addr().bits(addr));
309            }
310            self.qspi
311                .instrctrl()
312                .modify(|_, w| w.instr().bits(command.bits()));
313            self.qspi.instrframe().write(|w| {
314                tfm.instrframe(
315                    w,
316                    if command == Command::QuadPageProgram {
317                        instrframe::Tfrtypeselect::Writememory
318                    } else {
319                        instrframe::Tfrtypeselect::Write
320                    },
321                )
322            });
323            self.qspi.instrframe().read().bits();
324
325            if !buf.is_empty() {
326                core::ptr::copy(buf.as_ptr(), (QSPI_AHB + addr) as *mut u8, buf.len());
327            }
328
329            self.finalize();
330        }
331    }
332
333    unsafe fn run_read_instruction(
334        &self,
335        command: Command,
336        tfm: TransferMode,
337        addr: u32,
338        buf: &mut [u8],
339        finalize: bool,
340    ) {
341        unsafe {
342            self.qspi
343                .instrctrl()
344                .modify(|_, w| w.instr().bits(command.bits()));
345            self.qspi.instrframe().write(|w| {
346                tfm.instrframe(
347                    w,
348                    if command == Command::QuadRead {
349                        instrframe::Tfrtypeselect::Readmemory
350                    } else {
351                        instrframe::Tfrtypeselect::Read
352                    },
353                )
354            });
355            self.qspi.instrframe().read().bits();
356
357            if !buf.is_empty() {
358                core::ptr::copy((QSPI_AHB + addr) as *mut u8, buf.as_mut_ptr(), buf.len());
359            }
360
361            if finalize {
362                self.finalize();
363            }
364        }
365    }

367    /// Set the clock divider, relative to the main clock
368    ///
369    /// This fn safely subtracts 1 from your input value as the underlying fn is
370    /// SCK Baud = MCKL / (value + 1)
371    ///
372    /// ex if Mclk is 120mhz
373    /// value  0 is reduced to  0 results in 120mhz clock
374    /// value  1 is reduced to  0 results in 120mhz clock
375    /// value  2 is reduced to  1 results in  60mhz clock
376    pub fn set_clk_divider(&mut self, value: u8) {
377        // The baud register is divisor - 1
378        self.qspi
379            .baud()
380            .write(|w| unsafe { w.baud().bits(value.saturating_sub(1)) });
381    }
382}

384#[derive(Default, Debug, Copy, Clone)]
385struct TransferMode {
386    quad_width: bool,
387    data_enable: bool,
388    opcode_enable: bool,
389    address_enable: bool,
390    instruction_enable: bool,
391    dummy_cycles: u8,
392}
393
394impl TransferMode {
395    unsafe fn instrframe(
396        self,
397        instrframe: &mut instrframe::W,
398        tfrtype: instrframe::Tfrtypeselect,
399    ) -> &mut instrframe::W {
400        if self.quad_width {
401            instrframe.width().quad_output();
402        } else {
403            instrframe.width().single_bit_spi();
404        }
405
406        if self.data_enable {
407            instrframe.dataen().set_bit();
408        }
409        if self.opcode_enable {
410            instrframe.dataen().set_bit();
411        }
412        if self.address_enable {
413            instrframe.addren().set_bit();
414        }
415        if self.instruction_enable {
416            instrframe.instren().set_bit();
417        }
418
419        if self.dummy_cycles > 0 {
420            unsafe {
421                instrframe.dummylen().bits(self.dummy_cycles);
422            }
423        }
424        instrframe.addrlen()._24bits();
425        instrframe.optcodeen().clear_bit();
426        instrframe.tfrtype().variant(tfrtype);
427        instrframe
428    }
429}
