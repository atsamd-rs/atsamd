9/// Container for a set of SERCOM [`Pad`]s
10///

12/// details on specifying a `Pads` type and creating instances.

15    S: Sercom,
16    I: IoSet,
17    SDA: IsI2cPad<PadNum = Pad0, Sercom = S> + InIoSet<I>,
18    SCL: IsI2cPad<PadNum = Pad1, Sercom = S> + InIoSet<I>,
19{
20    sercom: PhantomData<S>,
21    ioset: PhantomData<I>,
22    sda: SDA,
23    scl: SCL,
24}

64/// Define a set of [`Pads`] using [`PinId`]s instead of [`Pin`]s
65///
66/// In some cases, it is more convenient to specify a set of `Pads` using
67/// `PinId`s rather than `Pin`s. This alias makes it easier to do so.
68///
69/// The first two parameters are the [`Sercom`] and [`IoSet`], while the
70/// remaining two are [`PinId`]s representing the corresponding type
71/// parameters of [`Pads`], i.e. `SDA` & `SCL`.
72///
73/// ```
74/// use atsamd_hal::pac::Peripherals;
75/// use atsamd_hal::gpio::{PA08, PA09, Pins};
76/// use atsamd_hal::sercom::{Sercom0, IoSet1, i2c};
77/// use atsamd_hal::typelevel::NoneT;
78///
79/// pub type Pads = i2c::PadsFromIds<Sercom0, IoSet1, PA08, PA09>;
80///
81/// pub fn create_pads() -> Pads {
82///     let peripherals = Peripherals::take().unwrap();
83///     let pins = Pins::new(peripherals.PORT);
84///     i2c::Pads::default().sda(pins.pa08).scl(pins.pa09)
85/// }
86/// ```
87///
88/// [`Pin`]: crate::gpio::Pin
89/// [`PinId`]: crate::gpio::PinId
90pub type PadsFromIds<S, I, SDA, SCL> = Pads<S, I, Pad<S, SDA>, Pad<S, SCL>>;

96/// Type-level function to recover the [`Pad`] types from a generic set
97/// of [`Pads`]
98///
99/// This trait is used as an interface between the [`Pads`] type and other
100/// types in this module. It acts as a [type-level function], returning the
101/// corresponding [`Sercom`] and [`OptionalPad`] types. It serves to cut down on
102/// the total number of type parameters needed in the [`Config`] struct. The
103/// `Config` struct doesn't need access to the [`Pin`]s directly.  Rather, it
104/// only needs to apply the [`SomePad`] trait bound when a `Pin` is required.
105/// The `PadSet` trait allows each `Config` struct to store an instance of
106/// `Pads` without itself being generic over all six type parameters of the
107/// `Pads` type.
108///
109/// This trait is a simplified version of the [`AnyKind`] trait pattern.
110///
111/// [`Pin`]: crate::gpio::Pin
112/// [`Config`]: super::Config
113/// [type-level function]: crate::typelevel#type-level-functions
114/// [`AnyKind`]: crate::typelevel#anykind-trait-pattern
115pub trait PadSet: Sealed {
116    type Sercom: Sercom;
117    type IoSet: IoSet;
118    type Sda: IsI2cPad<PadNum = Pad0, Sercom = Self::Sercom> + InIoSet<Self::IoSet>;
119    type Scl: IsI2cPad<PadNum = Pad1, Sercom = Self::Sercom> + InIoSet<Self::IoSet>;
120}
