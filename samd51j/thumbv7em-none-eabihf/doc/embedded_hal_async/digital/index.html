/// Asynchronously wait until the `ready_pin` becomes high.
async fn wait_until_ready<P>(ready_pin: &mut P)
where
    P: Wait,
{
    ready_pin
        .wait_for_high()
        .await
        .expect("failed to await input pin")
}
