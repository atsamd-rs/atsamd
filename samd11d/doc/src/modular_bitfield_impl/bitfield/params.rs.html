8/// Raises an unsupported argument compile time error.
9fn unsupported_argument<T>(arg: T) -> syn::Error
10where
11    T: Spanned,
12{
13    format_err!(arg, "encountered unsupported #[bitfield] attribute")
14}

16/// The parameters given to the `#[bitfield]` proc. macro.
17///
18/// # Example
19///
20/// ```rust
21/// # use modular_bitfield::prelude::*;
22/// #
23/// #[bitfield(bytes = 4, filled = true)]
24/// #[derive(BitfieldSpecifier)]
25/// pub struct SignedInteger {
26///     sign: bool,
27///     value: B31,
28/// }
29/// ```
30pub struct ParamArgs {
31    args: syn::AttributeArgs,
32}
33
34impl syn::parse::Parse for ParamArgs {
35    fn parse(input: syn::parse::ParseStream) -> Result<Self> {
36        let punctuated =
37            <syn::punctuated::Punctuated<_, syn::Token![,]>>::parse_terminated(input)?;
38        Ok(Self {
39            args: punctuated.into_iter().collect::<Vec<_>>(),
40        })
41    }
42}
43
44impl IntoIterator for ParamArgs {
45    type Item = syn::NestedMeta;
46    type IntoIter = std::vec::IntoIter<syn::NestedMeta>;
47
48    fn into_iter(self) -> Self::IntoIter {
49        self.args.into_iter()
50    }
51}

53impl Config {
54    /// Feeds a parameter that takes an integer value to the `#[bitfield]` configuration.
55    fn feed_int_param<F>(
56        name_value: syn::MetaNameValue,
57        name: &str,
58        on_success: F,
59    ) -> Result<()>
60    where
61        F: FnOnce(usize, Span) -> Result<()>,
62    {
63        assert!(name_value.path.is_ident(name));
64        match &name_value.lit {
65            syn::Lit::Int(lit_int) => {
66                let span = lit_int.span();
67                let value = lit_int.base10_parse::<usize>().map_err(|err| {
68                    format_err!(
69                        span,
70                        "encountered malformatted integer value for `{}` parameter: {}",
71                        name,
72                        err
73                    )
74                })?;
75                on_success(value, name_value.span())?;
76            }
77            invalid => {
78                return Err(format_err!(
79                    invalid,
80                    "encountered invalid value argument for #[bitfield] `{}` parameter",
81                    name
82                ))
83            }
84        }
85        Ok(())
86    }
87
88    /// Feeds a `bytes: int` parameter to the `#[bitfield]` configuration.
89    fn feed_bytes_param(&mut self, name_value: syn::MetaNameValue) -> Result<()> {
90        Self::feed_int_param(name_value, "bytes", |value, span| self.bytes(value, span))
91    }
92
93    /// Feeds a `bytes: int` parameter to the `#[bitfield]` configuration.
94    fn feed_bits_param(&mut self, name_value: syn::MetaNameValue) -> Result<()> {
95        Self::feed_int_param(name_value, "bits", |value, span| self.bits(value, span))
96    }
97
98    /// Feeds a `filled: bool` parameter to the `#[bitfield]` configuration.
99    fn feed_filled_param(&mut self, name_value: syn::MetaNameValue) -> Result<()> {
100        assert!(name_value.path.is_ident("filled"));
101        match &name_value.lit {
102            syn::Lit::Bool(lit_bool) => {
103                self.filled(lit_bool.value, name_value.span())?;
104            }
105            invalid => {
106                return Err(format_err!(
107                invalid,
108                "encountered invalid value argument for #[bitfield] `filled` parameter",
109            ))
110            }
111        }
112        Ok(())
113    }
114
115    /// Feeds the given parameters to the `#[bitfield]` configuration.
116    ///
117    /// # Errors
118    ///
119    /// If a parameter is malformatted, unexpected, duplicate or in conflict.
120    pub fn feed_params<'a, P>(&mut self, params: P) -> Result<()>
121    where
122        P: IntoIterator<Item = syn::NestedMeta> + 'a,
123    {
124        for nested_meta in params {
125            match nested_meta {
126                syn::NestedMeta::Meta(meta) => {
127                    match meta {
128                        syn::Meta::NameValue(name_value) => {
129                            if name_value.path.is_ident("bytes") {
130                                self.feed_bytes_param(name_value)?;
131                            } else if name_value.path.is_ident("bits") {
132                                self.feed_bits_param(name_value)?;
133                            } else if name_value.path.is_ident("filled") {
134                                self.feed_filled_param(name_value)?;
135                            } else {
136                                return Err(unsupported_argument(name_value))
137                            }
138                        }
139                        unsupported => return Err(unsupported_argument(unsupported)),
140                    }
141                }
142                unsupported => return Err(unsupported_argument(unsupported)),
143            }
144        }
145        Ok(())
146    }
147}
