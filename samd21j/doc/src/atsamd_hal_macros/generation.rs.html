13fn lookup_peripheral(peripheral: &str) -> Result<&'static [&'static str], Error> {
14    if let Some(cur_devices) = PERIPHERALS.get(peripheral) {
15        Ok(*cur_devices)
16    } else {
17        Err(Error::UnknownPeripheral {
18            peripheral: peripheral.to_string(),
19        })
20    }
21}

23fn cfg_token_tree<'a>(devices: impl IntoIterator<Item = &'a str>) -> Group {
24    let cfg_args = cfg_args(devices);
25    let cfg_args = TokenTree::Group(Group::new(
26        Delimiter::Parenthesis,
27        cfg_args.into_iter().collect(),
28    ));
29
30    Group::new(
31        Delimiter::Bracket,
32        [
33            TokenTree::Ident(Ident::new("cfg", Span::call_site())),
34            cfg_args,
35        ]
36        .into_iter()
37        .collect(),
38    )
39}

41pub fn cfg_args<'a>(devices: impl IntoIterator<Item = &'a str>) -> Vec<TokenTree> {
42    let mut inner = Vec::new();
43    for device in devices {
44        if !inner.is_empty() {
45            inner.push(TokenTree::Punct(Punct::new(',', Spacing::Alone)));
46        }
47        inner.push(TokenTree::Ident(Ident::new("feature", Span::call_site())));
48        inner.push(TokenTree::Punct(Punct::new('=', Spacing::Alone)));
49        inner.push(TokenTree::Literal(Literal::string(device)));
50    }
51    let any_args = TokenTree::Group(Group::new(
52        Delimiter::Parenthesis,
53        inner.into_iter().collect(),
54    ));
55    vec![
56        TokenTree::Ident(Ident::new("any", Span::call_site())),
57        any_args,
58    ]
59}

61pub fn hal_expr_to_devices(expr: &HalExpr) -> Result<BTreeSet<&'static str>, Error> {
62    match expr {
63        HalExpr::All(args) => {
64            let mut devices = all_devices();
65            for arg in args {
66                devices = devices
67                    .intersection(&hal_expr_to_devices(arg)?)
68                    .copied()
69                    .collect();
70            }
71            Ok(devices)
72        }
73        HalExpr::Any(args) => {
74            let mut devices = BTreeSet::new();
75            for arg in args {
76                devices.extend(hal_expr_to_devices(arg)?);
77            }
78            Ok(devices)
79        }
80        HalExpr::Not(arg) => {
81            let mut devices = all_devices();
82            for device in hal_expr_to_devices(arg)? {
83                devices.remove(device);
84            }
85            Ok(devices)
86        }
87        HalExpr::Peripheral(lit) => Ok(lookup_peripheral(lit)?.iter().copied().collect()),
88    }
89}

91pub fn gen_cfgs(expr: &HalExpr) -> Result<Group, Error> {
92    Ok(cfg_token_tree(hal_expr_to_devices(expr)?))
93}

95pub fn add_cfgs_to_input(cfgs: Group, input: TokenStream) -> TokenStream {
96    [
97        TokenTree::Punct(Punct::new('#', Spacing::Alone)),

99    ]
100    .into_iter()
101    .chain(input)
102    .collect::<TokenStream>()
